<script>
function markComplete(box,id){

    let dateSpan=document.getElementById(id);

    if(box.checked){
        let date=new Date().toLocaleDateString();
        dateSpan.innerHTML="Completed: "+date;
        localStorage.setItem(id,date);
    }else{
        dateSpan.innerHTML="";
        localStorage.removeItem(id);
    }

    updateProgress();
}

function updateProgress(){

    let total=document.querySelectorAll("input[type='checkbox']").length;
    let completed=document.querySelectorAll("input[type='checkbox']:checked").length;

    let percent=Math.round((completed/total)*100);

    let bar=document.getElementById("progressBar");

    bar.style.width=percent+"%";
    bar.innerHTML=percent+"%";
}

window.onload=function(){

    document.querySelectorAll("input[type='checkbox']").forEach(function(box){

        let id=box.getAttribute("onchange")
        .match(/'([^']+)'/)[1];

        let savedDate=localStorage.getItem(id);

        if(savedDate){
            box.checked=true;
            document.getElementById(id).innerHTML=
            "Completed: "+savedDate;
        }
    });

    updateProgress();
}
